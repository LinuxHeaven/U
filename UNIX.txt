3. The three sides of a triangle are given through the keyboard. Write program to 
calculate the area and perimeter of the triangle. Also check whether it is possible to 
construct a triangle with the input provided.  
#!/bin/bash 
echo "Enter three sides:" 
read a b c 
if [ $((a+b)) -gt $c ] && [ $((a+c)) -gt $b ] && [ $((b+c)) -gt $a ] 
then 
    perimeter=$((a+b+c)) 
    s=$((perimeter/2)) 
    area=$(echo "scale=2; sqrt($s*($s-$a)*($s-$b)*($s-$c))" | bc) 
    echo "Triangle is valid" 
    echo "Perimeter = $perimeter" 
    echo "Area = $area" 
else 
    echo "Triangle cannot be formed" 
fi

2. Write a C program to create an orphan process. Also explain which process takes 
responsibility of the orphan process.  
#include <stdio.h> 
#include <unistd.h> 
#include <stdlib.h> 
#include <sys/types.h> 
#include <sys/wait.h> 
int main(void) 
{ 
    int pid = 0; 
    int status; 
    pid = fork(); 
    if(pid == 0) 
    { 
        printf("I am the child.\n"); 
        sleep(10); 
        printf("I am the child, 10 seconds later.\n"); 
    } 
    if(pid > 0) 
    { 
        printf("I am the parent, the child is %d.\n", pid); 
        pid = wait(&status); 
        printf("End of process %d.\n", pid); 
        if(WIFEXITED(status)) 
        { 
            printf("The process ended with exit status: %d.\n", 
                   WEXITSTATUS(status)); 
        } 
        if(WIFSIGNALED(status)) 
        { 
            printf("The process ended with kill signal: %d.\n", 
                   WTERMSIG(status)); 
        } 
    } 
    if(pid < 0) 
    { 
        perror("In fork()"); 
    } 
    exit(0); 
    return 0; 
}

4. Write a C program using fork() system call to show how a file can be shared between 
processes.  
#include <stdio.h> 
#include <unistd.h> 
#include <stdlib.h> 
#include <sys/types.h> 
#include <sys/wait.h> 
#include <fcntl.h> 
int main() 
{ 
    int fd = open("shared_file.txt", 
                  O_WRONLY | O_CREAT | O_TRUNC, 
                  0666); 
    if(fd == -1) 
    { 
        perror("open"); 
        exit(EXIT_FAILURE); 
    } 
    pid_t pid = fork(); 
    if(pid == -1) 
    { 
        perror("fork"); 
        exit(EXIT_FAILURE); 
    } 
    else if(pid == 0) 
    { 
        printf("Child process writing to file\n"); 
        dprintf(fd, "Hello from child process\n"); 
        close(fd); 
        exit(EXIT_SUCCESS); 
    } 
    else 
    { 
        printf("Parent process waiting for child to finish\n"); 
        wait(NULL); 
        printf("Parent process writing to file\n"); 
        dprintf(fd, "Hello from parent process\n"); 
        close(fd); 
    } 
    return 0; 
}

    } 
    return 0; 
}

3. Write a program to create a child process. Send SIGCLD signal after 5 sec. 
#include <stdio.h> 
#include <unistd.h> 
#include <signal.h> 
#include <sys/wait.h> 
void sigcld_handler(int sig) 
{ 
    printf("SIGCLD Signal Received\n"); 
} 
int main() 
{ 
    signal(SIGCLD, sigcld_handler); 
    pid_t pid = fork(); 
    if(pid == 0) 
    { 
        printf("Child Process Running...\n"); 
        sleep(5); 
        printf("Child Process Exiting...\n"); 
    } 
    else 
    { 
        wait(NULL); 
    } 
    return 0; 
} 





1. Write a program for reading output from an external program using the popen() 
function.  
#include <stdio.h> 
int main() 
{ 
    FILE *fp; 
    char buffer[100]; 
    fp = popen("date", "r"); 
    while(fgets(buffer, sizeof(buffer), fp) != NULL) 
    { 
        printf("%s", buffer); 
    } 
    pclose(fp); 
    return 0; 
}

4. Write a program using pipe system call to communicate messages among two 
programs. 
#include <stdio.h> 
#include <unistd.h> 
#include <string.h> 
int main() 
{ 
    int fd[2]; 
    char msg[] = "Hello from Parent"; 
    char buffer[100]; 
    pipe(fd); 
    if(fork() == 0) 
    { 
        close(fd[1]); 
        read(fd[0], buffer, sizeof(buffer)); 
        printf("Child Received: %s\n", buffer); 
        close(fd[0]); 
    } 
    else 
    { 
        close(fd[0]); 
        write(fd[1], msg, strlen(msg)+1); 
        close(fd[1]); 
    } 
    return 0; 
}
